# 🧱 4. NOVA TABELA (ESSENCIAL)

```sql id="outbox1"
CREATE TABLE outbox_events (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    empresa_id VARCHAR(120) NOT NULL,
    event_type VARCHAR(80) NOT NULL,
    aggregate_type VARCHAR(80) NOT NULL,
    aggregate_id VARCHAR(120) NOT NULL,
    payload TEXT NOT NULL,
    status VARCHAR(20) NOT NULL,
    attempts INT NOT NULL DEFAULT 0,
    next_attempt_at TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE NOT NULL
);

CREATE INDEX idx_outbox_pending
ON outbox_events (status, next_attempt_at, empresa_id);
```

# ⚙️ 8. PROCESSADOR (substitui parte do ResultDeliveryProcessor)

```java id="outbox3"
@Component
public class OutboxProcessor {

    private final OutboxRepository repository;
    private final ResultWebhookClient webhookClient;

    @Scheduled(fixedDelay = 5000)
    public void process() {

        List<OutboxEventEntity> events =
            repository.findPending(Instant.now());

        for (var event : events) {
            try {
                if (event.getEventType().equals("RESULT_READY")) {

                    TestResultResponse payload =
                        parse(event.getPayload());

                    webhookClient.postResult(
                        resolveWebhook(event),
                        payload
                    );
                }

                event.setStatus("SENT");

            } catch (Exception ex) {
                event.setAttempts(event.getAttempts() + 1);
                event.setStatus(event.getAttempts() > 5 ? "DLQ" : "RETRYING");
            }
        }
    }
}
```
